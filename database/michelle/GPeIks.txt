% Parameters:
gks = [1]
Ek = [-100]

 
% Functions:
GPeIks(V,m,h) = -gks.*(m.^4).*h.*(V-Ek)

minf(V) = ((1.0) ./ (1 + exp((-33.2 - V)/9.1)))
hinf(V) = 0.2 + ((1.0 - 0.2) ./ (1 + exp(-(-20.0 - V)/10.0)))

taum(V) = 0.1 + ((30.0 - 0.1) ./ (exp((-33.2-V)/21.7) + exp(-(-33.2-V)/13.9)))
tauh = 3400.0


% ODEs:
m' = (minf(V)-m)./taum(V)
h' = (hinf(V)-h)./tauh


% Initial Conditions
m(0) = zeros(Npop,1)
h(0) = zeros(Npop,1)

 
% Substitution:
current => GPeIks(V,m,h)
